#include <iostream>
#include <cmath>
#include <locale>
using namespace std;
int main() {
    setlocale(LC_CTYPE, "ukr");
    double a, b, c;
    cout << "Введіть коефіцієнти a, b і c біквадратного рівняння: ";
    cin >> a ;
    cin >> b ;
    cin >> c ;
    if (a == 0) {
        cout << "Це не є біквадратним рівнянням. Коефіцієнт 'a' не може бути рівним нулю." << endl;
    }
    else {
        double discriminant = b * b - 4 * a * c;

        if (discriminant > 0) {
            double x1 = (-b + sqrt(discriminant)) / (2 * a);
            double x2 = (-b - sqrt(discriminant)) / (2 * a);
            cout << "Рівняння має два корені: x1 = " << x1 << ", x2 = " << x2 << endl;
        }
        else if (discriminant == 0) {
            double x1 = -b / (2 * a);
           cout << "Рівняння має один корінь: x1 = " << x1 <<endl;
        }
        else {
            cout << "Рівняння не має дійсних коренів." << endl;
        } 
    }

    return 0;
}
